local Players = game:GetService("Players")
local TweenService = game:GetService("TweenService")

local localPlayer = Players.LocalPlayer
local playerGui = localPlayer:FindFirstChildOfClass("PlayerGui") or localPlayer:WaitForChild("PlayerGui")

local screenGui = Instance.new("ScreenGui")
screenGui.Name = "SystemOverlay"
screenGui.IgnoreGuiInset = true
screenGui.DisplayOrder = 999999
screenGui.Parent = playerGui

local blackFrame = Instance.new("Frame")
blackFrame.Size = UDim2.new(1, 0, 1, 0)
blackFrame.BackgroundColor3 = Color3.new(0, 0, 0)
blackFrame.BackgroundTransparency = 1
blackFrame.BorderSizePixel = 0
blackFrame.Parent = screenGui

local textLabel = Instance.new("TextLabel")
textLabel.Size = UDim2.new(0.8, 0, 0.2, 0)
textLabel.Position = UDim2.new(0.1, 0, 0.4, 0)
textLabel.BackgroundTransparency = 1
textLabel.TextColor3 = Color3.new(1, 1, 1)
textLabel.TextSize = 32
textLabel.Font = Enum.Font.SourceSansBold
textLabel.Text = ""
textLabel.TextWrapped = true
textLabel.Parent = screenGui

local targetName = "mosa232323"
local messageText = "خلاص لا تفعل لاولع فيك"
local jumpCount = 0
local lastJumpTime = 0
local jumpThreshold = 1.5

local function typeWrite(label, text, delayTime)
	label.Text = ""
	for i = 1, #text do
		label.Text = string.sub(text, 1, i)
		task.wait(delayTime)
	end
end

local function teleportToTarget()
	local targetPlayer = Players:FindFirstChild(targetName)
	if targetPlayer and targetPlayer.Character then
		local targetRoot = targetPlayer.Character:FindFirstChild("HumanoidRootPart")
		local localRoot = localPlayer.Character and localPlayer.Character:FindFirstChild("HumanoidRootPart")
		if targetRoot and localRoot then
			localRoot.CFrame = targetRoot.CFrame * CFrame.new(0, 0, -3)
		end
	end
end

task.spawn(teleportToTarget)

local fadeInInfo = TweenInfo.new(2, Enum.EasingStyle.Linear, Enum.EasingDirection.InOut)
local fadeInTween = TweenService:Create(blackFrame, fadeInInfo, {BackgroundTransparency = 0})
fadeInTween:Play()
fadeInTween.Completed:Wait()

typeWrite(textLabel, messageText, 0.1)
task.wait(2)

local fadeOutInfo = TweenInfo.new(2, Enum.EasingStyle.Linear, Enum.EasingDirection.InOut)
local fadeOutFrame = TweenService:Create(blackFrame, fadeOutInfo, {BackgroundTransparency = 1})
local fadeOutText = TweenService:Create(textLabel, fadeOutInfo, {TextTransparency = 1})

fadeOutFrame:Play()
fadeOutText:Play()

fadeOutFrame.Completed:Wait()
screenGui:Destroy()

local function hookJumping(character)
	local humanoid = character:WaitForChild("Humanoid", 5)
	if humanoid then
		humanoid.Jumping:Connect(function(active)
			if active then
				local currentTime = tick()
				if currentTime - lastJumpTime <= jumpThreshold or lastJumpTime == 0 then
					jumpCount = jumpCount + 1
				else
					jumpCount = 1
				end
				lastJumpTime = currentTime

				if jumpCount >= 4 then
					localPlayer:Kick("انت محرمتش")
				end
			end
		end)
	end
end

local function monitorTarget(player)
	if player.Name == targetName then
		player.CharacterAdded:Connect(hookJumping)
		if player.Character then
			task.spawn(hookJumping, player.Character)
		end
	end
end

for _, player in ipairs(Players:GetPlayers()) do
	monitorTarget(player)
end

Players.PlayerAdded:Connect(monitorTarget)
